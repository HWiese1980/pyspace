[docs]def factorize(n):
    """
    Simple function for prime factorization using a recursive approach.
    Suitable for small n.
    """
    prime_factors = []
    result = n
    
    # 1 is always a factor...
    if n == 1:
        return [1]
    
    while True:
        if result == 1:
            break
        divisor = 2
        while True:
            if result % divisor == 0:
                break
            divisor += 1
        prime_factors.append(divisor)
        result /= divisor
    return prime_factors
